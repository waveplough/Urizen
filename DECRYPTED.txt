set numbers {10 20 30 40 50}
set sum 0
foreach num $numbers {
    set sum [expr {$sum + $num}]
}
puts "The sum of the numbers is: $sum"

proc factorial {n} {
    if {$n <= 1} {
        return 1
    } else {
        return [expr {$n * [factorial [expr {$n - 1}]]}]
    }
}

set fact5 [factorial 5]
puts "Factorial of 5 is: $fact5"
